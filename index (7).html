import {$,num,fmtDate,requireAuth,loadCollection,loadSettings,userCollection,userDoc,drawChart} from './common.js';import {addDoc,deleteDoc,serverTimestamp} from './firebase.js'; let user,entries=[],settings;function currentWeight(){return entries.filter(x=>x.type==='weight').sort((a,b)=>String(b.date).localeCompare(String(a.date)))[0]}function latestMeasure(){return entries.filter(x=>x.type==='measurements').sort((a,b)=>String(b.date).localeCompare(String(a.date)))[0]}function render(){const w=currentWeight(),m=latestMeasure(),bmi=w&&settings.heightCm?num(w.weight)/((num(settings.heightCm)/100)**2):null;$('bodySummary').innerHTML=`
⚖️${w?`${num(w.weight)} ${settings.unit}`:'—'}Current weight
📐${settings.heightCm||'—'} cmHeight
📊${bmi?bmi.toFixed(1):'—'}BMI
📏${m?fmtDate(m.date):'—'}Latest measurements
`;renderChart();renderHistory();}function renderChart(){const metric=$('chartMetric').value;const points=entries.filter(x=>metric==='weight'?x.type==='weight':x.type==='measurements'&&x[metric]!==undefined).sort((a,b)=>String(a.date).localeCompare(String(b.date))).map(x=>({date:x.date,value:metric==='weight'?num(x.weight):num(x[metric])}));drawChart($('bodyChart'),points)}function renderHistory(){$('bodyHistoryList').innerHTML=entries.sort((a,b)=>String(b.date).localeCompare(String(a.date))).map(x=>`
${x.type==='weight'?'Weight':'Measurements'}
${fmtDate(x.date)} · ${x.type==='weight'?`${num(x.weight)} ${settings.unit}`:`Waist ${x.waist||'—'} · Hips ${x.hips||'—'}`}
`).join('')||'

No entries yet.

';} function open(type){$('bodyType').value=type;$('bodyDate').value=new Date().toISOString().slice(0,10);$('bodyDialogTitle').textContent=type==='weight'?'Add weight':'Add measurements';$('weightFields').classList.toggle('hidden',type!=='weight');$('measureFields').classList.toggle('hidden',type==='weight');$('bodyDialog').showModal();} requireAuth(async u=>{user=u;[entries,settings]=await Promise.all([loadCollection(u.uid,'bodyEntries'),loadSettings(u.uid)]);render();$('addWeightBtn').onclick=()=>open('weight');$('addMeasureBtn').onclick=()=>open('measurements');$('historyBtn').onclick=()=>$('bodyHistoryDialog').showModal();$('closeBodyHistory').onclick=()=>$('bodyHistoryDialog').close();$('chartMetric').onchange=renderChart;$('saveBodyEntry').onclick=async e=>{e.preventDefault();const type=$('bodyType').value,data={type,date:$('bodyDate').value,createdAt:serverTimestamp()};if(type==='weight')data.weight=num($('bodyWeight').value);else for(const f of ['waist','hips','chest','leftArm','rightArm','leftLeg','rightLeg'])data[f]=num($(f).value);const ref=await addDoc(userCollection(u.uid,'bodyEntries'),data);entries.push({id:ref.id,...data});$('bodyDialog').close();render();};$('bodyHistoryList').onclick=async e=>{if(!e.target.dataset.delete)return;if(confirm('Delete this entry?')){await deleteDoc(userDoc(u.uid,'bodyEntries',e.target.dataset.delete));entries=entries.filter(x=>x.id!==e.target.dataset.delete);render();}};});
